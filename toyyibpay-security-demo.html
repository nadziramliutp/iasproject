// Client-side encryption before transmission
async function encryptPaymentData(paymentData) {
    const key = await crypto.subtle.generateKey(
        { name: 'AES-GCM', length: 256 },
        true,
        ['encrypt', 'decrypt']
    );
    
    const iv = crypto.getRandomValues(new Uint8Array(12));
    const encodedData = new TextEncoder().encode(
        JSON.stringify(paymentData)
    );
    
    const encryptedData = await crypto.subtle.encrypt(
        { name: 'AES-GCM', iv: iv },
        key,
        encodedData
    );
    
    return {
        encryptedData: new Uint8Array(encryptedData),
        iv: iv,
        keyId: 'key_' + Math.random().toString(36).substr(2, 9)
    };
}

class PaymentEncryption:
    def __init__(self):
        self.key = Fernet.generate_key()
        self.cipher = Fernet(self.key)
    
    def encrypt_payment_data(self, payment_data):
        """Encrypt sensitive payment information"""
        json_data = json.dumps(payment_data).encode()
        encrypted_data = self.cipher.encrypt(json_data)
        
        return {
            'encrypted_payload': encrypted_data.decode(),
            'key_id': self.generate_key_id(),
            'timestamp': int(time.time())
        }
    
    def tokenize_card(self, card_number):
        """Replace card number with secure token"""
        token = 'tok_' + secrets.token_urlsafe(16)
        
        # Store in secure vault (Redis/Database)
        self.store_token_mapping(token, card_number)
        
        return {
            'token': token,
            'masked_card': '****-****-****-' + card_number[-4:]
        }

public class PaymentSecurityService {
    private static final String ALGORITHM = "AES/GCM/NoPadding";
    
    public EncryptedPayload encryptPaymentData(PaymentData data) 
            throws Exception {
        KeyGenerator keyGen = KeyGenerator.getInstance("AES");
        keyGen.init(256);
        SecretKey secretKey = keyGen.generateKey();
        
        Cipher cipher = Cipher.getInstance(ALGORITHM);
        cipher.init(Cipher.ENCRYPT_MODE, secretKey);
        
        byte[] encryptedData = cipher.doFinal(
            data.toJson().getBytes("UTF-8")
        );

            encryptedData,
            cipher.getIV(),

const jwt = require('jsonwebtoken');
const speakeasy = require('speakeasy');
const QRCode = require('qrcode');

class AuthenticationService {
    // Generate MFA secret for new user
    generateMFASecret(userEmail) {
        const secret = speakeasy.generateSecret({
            name: userEmail,
            issuer: 'SecurePayGateway',
            length: 32
        });
        
        return {
            secret: secret.base32,
            qrCode: secret.otpauth_url,
            backupCodes: this.generateBackupCodes()
        };
    }
    
    // Verify MFA token
    verifyMFAToken(secret, token) {
        return speakeasy.verify({
            secret: secret,
            encoding: 'base32',
            token: token,
            window: 2 // Allow 2-step tolerance
        });
    }
    
    // Generate OAuth 2.0 access token
    generateAccessToken(user, scopes) {
        const payload = {
            sub: user.id,
            email: user.email,
            scopes: scopes,
            iat: Math.floor(Date.now() / 1000),
            exp: Math.floor(Date.now() / 1000) + 3600 // 1 hour
        };
        
        return jwt.sign(payload, process.env.JWT_SECRET, {
            algorithm: 'HS256'
        });
    }
    
    // Session management with Redis
    async createSession(userId, deviceInfo) {
        const sessionId = crypto.randomUUID();
        const sessionData = {
            userId: userId,
            deviceInfo: deviceInfo,
            createdAt: new Date(),
            lastActivity: new Date(),
            isActive: true
        };
        
        // Store in Redis with TTL
        await redis.setex(
            `session:${sessionId}`, 
            1800, // 30 minutes
            JSON.stringify(sessionData)
        );
        
        return sessionId;
    }
}

class KYCVerificationService:
    def __init__(self):
        self.api_key = os.getenv('DOCUMENT_VERIFICATION_API_KEY')
        self.base_url = 'https://api.documentverification.service'
    
    async def verify_business_registration(self, document_image):
        """Verify business registration certificate"""
        # Extract text using OCR
        extracted_text = pytesseract.image_to_string(
            cv2.imread(document_image)
        )
        
        # Check for required fields
        required_fields = [
            'company registration number',
            'date of incorporation',
            'registered address'
        ]
        
        verification_results = {}
        for field in required_fields:
            verification_results[field] = self.extract_field(
                extracted_text, field
            )
        
        # Validate against government database
        return await self.validate_with_authorities(verification_results)
    
    def verify_identity_document(self, id_image, selfie_image):
        """Verify ID document and match with selfie"""
        # Face recognition comparison
        id_face = face_recognition.load_image_file(id_image)
        selfie_face = face_recognition.load_image_file(selfie_image)
        
        id_encoding = face_recognition.face_encodings(id_face)[0]
        selfie_encoding = face_recognition.face_encodings(selfie_face)[0]
        
        # Compare faces
        face_match = face_recognition.compare_faces(
            [id_encoding], selfie_encoding, tolerance=0.6
        )[0]
        
        confidence = face_recognition.face_distance(
            [id_encoding], selfie_encoding
        )[0]
        
        return {
            'face_match': face_match,
            'confidence_score': 1 - confidence,
            'verification_status': 'verified' if face_match else 'failed'
        }
    
    async def comprehensive_kyc_check(self, merchant_data):
        """Complete KYC verification process"""
        results = {
            'business_verification': await self.verify_business_registration(
                merchant_data['business_doc']
            ),
            'identity_verification': self.verify_identity_document(
                merchant_data['id_document'],
                merchant_data['selfie']
            ),
            'risk_assessment': await self.assess_risk_profile(merchant_data),
            'sanctions_check': await self.check_sanctions_list(
                merchant_data['owner_name']
            )
        }
        
        # Calculate overall verification score
        overall_score = self.calculate_verification_score(results)
        
        return {
            'verification_results': results,
            'overall_score': overall_score,
            'status': 'approved' if overall_score > 0.8 else 'manual_review'
        }

package main

import (
    "crypto/sha256"
    "encoding/hex"
    "encoding/json"
    "time"
)

type Transaction struct {
    ID          string    `json:"id"`
    Amount      float64   `json:"amount"`
    From        string    `json:"from"`
    To          string    `json:"to"`
    Timestamp   time.Time `json:"timestamp"`
    PrevHash    string    `json:"prev_hash"`
    Hash        string    `json:"hash"`
}

type TransactionChain struct {
    Transactions []Transaction `json:"transactions"`
    LastHash     string        `json:"last_hash"`
}

// Generate cryptographic hash for transaction
func (t *Transaction) GenerateHash() string {
    data := struct {
        ID        string    `json:"id"`
        Amount    float64   `json:"amount"`
        From      string    `json:"from"`
        To        string    `json:"to"`
        Timestamp time.Time `json:"timestamp"`
        PrevHash  string    `json:"prev_hash"`
    }{
        ID:        t.ID,
        Amount:    t.Amount,
        From:      t.From,
        To:        t.To,
        Timestamp: t.Timestamp,
        PrevHash:  t.PrevHash,
    }
    
    jsonData, _ := json.Marshal(data)
    hash := sha256.Sum256(jsonData)
    return hex.EncodeToString(hash[:])
}

// Add transaction to immutable chain
func (tc *TransactionChain) AddTransaction(tx Transaction) {
    tx.PrevHash = tc.LastHash
    tx.Hash = tx.GenerateHash()
    
    tc.Transactions = append(tc.Transactions, tx)
    tc.LastHash = tx.Hash
}

// Verify entire transaction chain integrity
func (tc *TransactionChain) VerifyIntegrity() bool {
    for i := 1; i < len(tc.Transactions); i++ {
        current := tc.Transactions[i]
        previous := tc.Transactions[i-1]
        
        // Check if current transaction's prev_hash matches previous hash
        if current.PrevHash != previous.Hash {
            return false
        }
        
        // Verify current transaction hash
        if current.Hash != current.GenerateHash() {
            return false
        }
    }
    return true
}

// Immutable audit log service
type AuditService struct {
    logChain TransactionChain
}

func (as *AuditService) LogPaymentEvent(event PaymentEvent) {
    transaction := Transaction{
        ID:        generateUUID(),
        Amount:    event.Amount,
        From:      event.FromAccount,
        To:        event.ToAccount,
        Timestamp: time.Now(),
    }
    
    as.logChain.AddTransaction(transaction)
    
    // Store in append-only database
    as.persistToDatabase(transaction)
}

import asyncio
import redis
from datetime import datetime, timedelta
import json

class SecurityMonitoringService:
    def __init__(self):
        self.redis_client = redis.Redis(host='localhost', port=6379)
        self.alert_thresholds = {
            'failed_logins': 5,
            'api_rate_limit': 1000,
            'unusual_transaction_amount': 10000
        }
    
    async def monitor_failed_logins(self, ip_address, user_id):
        """Monitor failed login attempts"""
        key = f"failed_logins:{ip_address}"
        current_count = self.redis_client.incr(key)
        self.redis_client.expire(key, 3600)  # 1 hour window
        
        if current_count >= self.alert_thresholds['failed_logins']:
            await self.trigger_security_alert({
                'type': 'FAILED_LOGIN_THRESHOLD',
                'ip_address': ip_address,
                'user_id': user_id,
                'count': current_count,
                'severity': 'HIGH'
            })
            
            # Auto-block IP address
            await self.block_ip_address(ip_address, duration=3600)
    
    async def monitor_api_usage(self, api_key, endpoint):
        """Monitor API rate limiting"""
        key = f"api_usage:{api_key}:{endpoint}"
        usage_count = self.redis_client.incr(key)
        self.redis_client.expire(key, 3600)  # 1 hour window
        
        if usage_count >= self.alert_thresholds['api_rate_limit']:
            await self.trigger_security_alert({
                'type': 'API_RATE_LIMIT_EXCEEDED',
                'api_key': api_key,
                'endpoint': endpoint,
                'usage_count': usage_count,
                'severity': 'MEDIUM'
            })
            
            return False  # Block request
        
        return True  # Allow request
    
    async def detect_anomalous_transactions(self, transaction_data):
        """ML-based anomaly detection"""
        user_id = transaction_data['user_id']
        amount = transaction_data['amount']
        
        # Get user's transaction history
        history_key = f"user_tx_history:{user_id}"
        recent_transactions = self.redis_client.lrange(history_key, 0, 50)
        
        # Calculate user's average transaction amount
        if recent_transactions:
            amounts = [float(tx) for tx in recent_transactions]
            avg_amount = sum(amounts) / len(amounts)
            
            # Check if current transaction is significantly higher
            if amount > avg_amount * 5:  # 5x higher than average
                await self.trigger_security_alert({
                    'type': 'ANOMALOUS_TRANSACTION',
                    'user_id': user_id,
                    'amount': amount,
                    'avg_amount': avg_amount,
                    'severity': 'HIGH'
                })
                
                # Hold transaction for manual review
                return 'HOLD_FOR_REVIEW'
        
        # Store transaction in history
        self.redis_client.lpush(history_key, amount)
        self.redis_client.ltrim(history_key, 0, 99)  # Keep last 100
        
        return 'APPROVED'
    
    async def trigger_security_alert(self, alert_data):
        """Send security alerts to monitoring dashboard"""
        alert = {
            'timestamp': datetime.now().isoformat(),
            'alert_id': self.generate_alert_id(),
            **alert_data
        }
        
        # Store alert in database
        self.redis_client.lpush('security_alerts', json.dumps(alert))
        
        # Send to real-time dashboard via WebSocket
        await self.broadcast_alert(alert)
        
        # Send email/SMS for high severity alerts
        if alert_data['severity'] == 'HIGH':
            await self.send_urgent_notification(alert)

// Real-time security monitoring dashboard
class SecurityDashboard {
    constructor() {
        this.socket = new WebSocket('wss://api.securepaygateway.com/security-alerts');
        this.alertsContainer = document.getElementById('security-alerts');
        this.setupEventListeners();
    }
    
    setupEventListeners() {
        this.socket.onmessage = (event) => {
            const alert = JSON.parse(event.data);
            this.displayAlert(alert);
            this.updateMetrics(alert);
        };
        
        this.socket.onopen = () => {
            console.log('Connected to security monitoring');
            this.loadHistoricalAlerts();
        };
    }
    
    displayAlert(alert) {
        const alertElement = document.createElement('div');
        alertElement.className = `alert alert-${alert.severity.toLowerCase()}`;
        
        const timestamp = new Date(alert.timestamp).toLocaleString();
        
        alertElement.innerHTML = `
            <div class="alert-header">
                <span class="alert-type">${alert.type}</span>
                <span class="alert-time">${timestamp}</span>
            </div>
            <div class="alert-details">
                ${this.formatAlertDetails(alert)}
            </div>
            <div class="alert-actions">
                <button onclick="this.acknowledgeAlert('${alert.alert_id}')">
                    Acknowledge
                </button>
                <button onclick="this.blockSource('${alert.ip_address || alert.user_id}')">
                    Block Source
                </button>
            </div>
        `;
        
        this.alertsContainer.insertBefore(alertElement, this.alertsContainer.firstChild);
        
        // Play alert sound for high severity
        if (alert.severity === 'HIGH') {
            this.playAlertSound();
        }
    }
    
    formatAlertDetails(alert) {
        switch (alert.type) {
            case 'FAILED_LOGIN_THRESHOLD':
                return `
                    Multiple failed login attempts detected<br>
                    IP Address: ${alert.ip_address}<br>
                    Attempt Count: ${alert.count}
                `;
            
            case 'API_RATE_LIMIT_EXCEEDED':
                return `
                    API rate limit exceeded<br>
                    Endpoint: ${alert.endpoint}<br>
                    Request Count: ${alert.usage_count}
                `;
            
            case 'ANOMALOUS_TRANSACTION':
                return `
                    Unusual transaction detected<br>
                    Amount: RM ${alert.amount}<br>
                    User Average: RM ${alert.avg_amount}
                `;
            
            default:
                return 'Security event detected';
        }
    }
    
    async acknowledgeAlert(alertId) {
        try {
            await fetch(`/api/security/alerts/${alertId}/acknowledge`, {
                method: 'POST',
                headers: {
                    'Authorization': `Bearer ${this.getAuthToken()}`

            // Remove alert from display
            document.querySelector(`[data-alert-id="${alertId}"]`).remove();

            console.error('Failed to acknowledge alert:', error);
        }
    }
}
